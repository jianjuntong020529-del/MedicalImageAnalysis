```python
import nibabel as nib
import numpy as np
import SimpleITK as sitk

def read_nii_gz(nii_path):
    """读取nii.gz文件，返回像素数据和空间参数"""
    img = nib.load(nii_path)
    # 像素数据（H,W,D），D为切片数
    data = img.get_fdata().astype(np.int32)
    # 空间参数（原点、间距、方向矩阵）
    affine = img.affine  # 仿射矩阵（包含原点、间距、方向）
    header = img.header
    return data, affine, header

def save_nii_gz(data, affine, header, save_path):
    """保存修改后的分割掩码为nii.gz，保留原始空间参数"""
    # 确保数据类型与原始一致
    img = nib.Nifti1Image(data, affine=affine, header=header)
    nib.save(img, save_path)
    return True

def align_nii_with_original(seg_data, seg_affine, orig_affine):
    """对齐分割掩码与原始影像的空间参数（若不一致）"""
    # 简化版：若仿射矩阵不同，重采样分割掩码（完整版需用SimpleITK配准）
    if not np.allclose(seg_affine, orig_affine):
        # 此处仅示例，完整重采样需用SimpleITK
        print("分割掩码与原始影像空间参数不一致，已对齐")
    return seg_data
```

##### （2）分割编辑核心逻辑（src/core/segmentation_editor.py）
```python
import numpy as np
import cv2

class SegmentationEditor:
    def __init__(self):
        # 原始影像数据（H,W,D）
        self.orig_data = None
        # 分割掩码数据（H,W,D），与原始影像同尺寸
        self.seg_data = None
        # 空间参数
        self.orig_affine = None
        self.seg_affine = None
        self.seg_header = None
        # 编辑状态
        self.current_slice = 0  # 当前编辑切片
        self.current_label = 1  # 当前编辑标签（如1=肝脏）
        self.brush_size = 5     # 画笔大小
        self.alpha = 0.5        # 叠加透明度
        # 撤销/重做栈
        self.history_stack = []
        self.history_index = -1

    def load_data(self, orig_nii_path, seg_nii_path):
        """加载原始影像和分割掩码"""
        from src.utils.nii_utils import read_nii_gz, align_nii_with_original
        # 加载原始影像
        self.orig_data, self.orig_affine, _ = read_nii_gz(orig_nii_path)
        # 加载分割掩码
        self.seg_data, self.seg_affine, self.seg_header = read_nii_gz(seg_nii_path)
        # 对齐空间参数
        self.seg_data = align_nii_with_original(self.seg_data, self.seg_affine, self.orig_affine)
        # 初始化历史栈
        self.history_stack.append(self.seg_data.copy())
        self.history_index = 0
        return True

    def get_current_slice_data(self):
        """获取当前切片的原始影像和分割掩码"""
        orig_slice = self.orig_data[:, :, self.current_slice]
        seg_slice = self.seg_data[:, :, self.current_slice]
        return orig_slice, seg_slice

    def fuse_image(self, orig_slice, seg_slice):
        """融合原始影像和分割掩码（伪彩色叠加）"""
        # 原始影像归一化到0-255
        orig_norm = (orig_slice - orig_slice.min()) / (orig_slice.max() - orig_slice.min() + 1e-8) * 255
        orig_norm = orig_norm.astype(np.uint8)
        # 分割掩码伪彩色映射（单标签）
        seg_color = np.zeros((orig_norm.shape[0], orig_norm.shape[1], 3), dtype=np.uint8)
        # 标记当前编辑标签的区域为红色（可自定义颜色映射）
        seg_color[seg_slice == self.current_label] = [255, 0, 0]
        # 加权融合
        fused = cv2.addWeighted(cv2.cvtColor(orig_norm, cv2.COLOR_GRAY2RGB), 1-self.alpha, seg_color, self.alpha, 0)
        return fused

    def edit_seg_mask(self, x, y, mode="draw"):
        """
        编辑分割掩码（画笔/橡皮擦）
        :param x,y: 鼠标点击的像素坐标
        :param mode: draw（绘制）/erase（擦除）
        """
        # 保存当前状态到历史栈（清空重做栈）
        self.history_stack = self.history_stack[:self.history_index+1]
        # 生成画笔掩码（圆形）
        brush_mask = self._create_brush_mask(x, y)
        # 修改分割掩码
        if mode == "draw":
            self.seg_data[:, :, self.current_slice][brush_mask] = self.current_label
        elif mode == "erase":
            self.seg_data[:, :, self.current_slice][brush_mask] = 0
        # 加入历史栈
        self.history_stack.append(self.seg_data.copy())
        self.history_index += 1

    def _create_brush_mask(self, x, y):
        """生成圆形画笔掩码"""
        h, w = self.seg_data[:, :, self.current_slice].shape
        xx, yy = np.meshgrid(np.arange(w), np.arange(h))
        dist = np.sqrt((xx - x)**2 + (yy - y)**2)
        mask = dist <= self.brush_size
        return mask

    def undo(self):
        """撤销上一步操作"""
        if self.history_index > 0:
            self.history_index -= 1
            self.seg_data = self.history_stack[self.history_index].copy()
            return True
        return False

    def redo(self):
        """重做上一步操作"""
        if self.history_index < len(self.history_stack)-1:
            self.history_index += 1
            self.seg_data = self.history_stack[self.history_index].copy()
            return True
        return False

    def save_edited_seg(self, save_path):
        """保存修改后的分割掩码"""
        from src.utils.nii_utils import save_nii_gz
        return save_nii_gz(self.seg_data, self.orig_affine, self.seg_header, save_path)
```

##### （3）Qt界面实现（src/widgets/SegmentationEditWidget.py）
```python
import sys
import numpy as np
from PyQt6.QtWidgets import (
    QWidget, QVBoxLayout, QHBoxLayout, QPushButton, QSlider,
    QLabel, QSpinBox, QColorDialog, QFileDialog, QMessageBox
)
from PyQt6.QtCore import Qt, QPoint
from PyQt6.QtGui import QImage, QPixmap, QPainter, QColor

class SegmentationEditWidget(QWidget):
    def __init__(self):
        super().__init__()
        self.editor = None  # 分割编辑核心实例

        self.init_ui()

    def init_ui(self):
        # 主布局
        main_layout = QVBoxLayout(self)
        
        # 1. 工具栏
        tool_layout = QHBoxLayout()
        # 加载按钮
        self.load_btn = QPushButton("加载原始+分割影像")
        self.load_btn.clicked.connect(self.load_data)
        # 编辑模式切换
        self.draw_btn = QPushButton("画笔")
        self.draw_btn.setCheckable(True)
        self.erase_btn = QPushButton("橡皮擦")
        self.erase_btn.setCheckable(True)
        # 撤销/重做
        self.undo_btn = QPushButton("撤销")
        self.undo_btn.clicked.connect(self.on_undo)
        self.redo_btn = QPushButton("重做")

        self.save_btn = QPushButton("保存修改后的分割结果")

        tool_layout.addWidget(self.load_btn)
        tool_layout.addWidget(self.draw_btn)
        tool_layout.addWidget(self.erase_btn)
        tool_layout.addWidget(self.undo_btn)
        tool_layout.addWidget(self.redo_btn)
        tool_layout.addWidget(self.save_btn)
        main_layout.addLayout(tool_layout)
        
        # 2. 影像显示区（双窗格）
        view_layout = QHBoxLayout()
        # 原始影像显示
        self.orig_label = QLabel("原始影像")
        self.orig_label.setMinimumSize(400, 400)
        self.orig_label.setStyleSheet("border:1px solid #ccc;")
        self.orig_label.setAlignment(Qt.AlignmentFlag.AlignCenter)
        # 分割编辑显示
        self.edit_label = QLabel("分割编辑区")
        self.edit_label.setMinimumSize(400, 400)
        self.edit_label.setStyleSheet("border:1px solid #ccc;")
        self.edit_label.setAlignment(Qt.AlignmentFlag.AlignCenter)
        # 绑定鼠标事件
        self.edit_label.mousePressEvent = self.on_mouse_press
        self.edit_label.mouseMoveEvent = self.on_mouse_move
        self.edit_label.mouseReleaseEvent = self.on_mouse_release
        
        view_layout.addWidget(self.orig_label)
        view_layout.addWidget(self.edit_label)
        main_layout.addLayout(view_layout)
        
        # 3. 参数控制区
        param_layout = QHBoxLayout()
        # 切片控制
        param_layout.addWidget(QLabel("当前切片："))
        self.slice_slider = QSlider(Qt.Orientation.Horizontal)
        self.slice_slider.valueChanged.connect(self.on_slice_change)
        self.slice_label = QLabel("0/0")
        # 标签选择
        param_layout.addWidget(QLabel("编辑标签："))

    # ========== 事件处理 ==========
    def load_data(self):
        """加载原始影像和分割掩码"""
        # 选择文件
        orig_path, _ = QFileDialog.getOpenFileName(self, "选择原始影像（nii.gz）", "", "NIfTI Files (*.nii.gz *.nii)")
        seg_path, _ = QFileDialog.getOpenFileName(self, "选择分割掩码（nii.gz）", "", "NIfTI Files (*.nii.gz *.nii)")
        if not orig_path or not seg_path:
            return
        
        # 初始化编辑核心
        from src.core.segmentation_editor import SegmentationEditor
        self.editor = SegmentationEditor()
        try:
            self.editor.load_data(orig_path, seg_path)
        except Exception as e:
            QMessageBox.critical(self, "错误", f"加载失败：{str(e)}")

        slice_num = self.editor.orig_data.shape[2]
        self.slice_slider.setRange(0, slice_num-1)
        self.slice_label.setText(f"0/{slice_num-1}")

        self.update_image_display()

    def update_image_display(self):
        """更新影像显示"""
        if not self.editor:
            return
        
        # 获取当前切片数据
        orig_slice, seg_slice = self.editor.get_current_slice_data()
        # 融合分割掩码
        fused_slice = self.editor.fuse_image(orig_slice, seg_slice)
        
        # 显示原始影像
        orig_qimg = self.np2qimg(orig_slice)
        self.orig_label.setPixmap(QPixmap.fromImage(orig_qimg).scaled(
            self.orig_label.size(), Qt.AspectRatioMode.KeepAspectRatio, Qt.TransformationMode.SmoothTransformation
        ))
        # 显示融合后的编辑区
        edit_qimg = self.np2qimg(fused_slice)
        self.edit_label.setPixmap(QPixmap.fromImage(edit_qimg).scaled(
            self.edit_label.size(), Qt.AspectRatioMode.KeepAspectRatio, Qt.TransformationMode.SmoothTransformation
        ))

    def np2qimg(self, np_img):
        """numpy数组转QImage"""
        # 处理灰度图
        if len(np_img.shape) == 2:
            h, w = np_img.shape
            qimg = QImage(np_img.data, w, h, w, QImage.Format.Format_Grayscale8)
        # 处理彩色图
        elif len(np_img.shape) == 3:
            h, w, ch = np_img.shape
            qimg = QImage(np_img.data, w, h, w*ch, QImage.Format.Format_RGB888)
        else:
            return QImage()
        return qimg

    def on_slice_change(self, value):
        """切换切片"""
        if not self.editor:
            return
        self.editor.current_slice = value
        self.slice_label.setText(f"{value}/{self.editor.orig_data.shape[2]-1}")
        self.update_image_display()

    def on_label_change(self, value):
        """切换编辑标签"""
        if self.editor:
            self.editor.current_label = value

    def on_brush_size_change(self, value):
        """修改画笔大小"""
        if self.editor:
            self.editor.brush_size = value

    def on_alpha_change(self, value):
        """修改叠加透明度"""
        if self.editor:
            self.editor.alpha = value / 100.0
            self.update_image_display()

    def on_undo(self):
        """撤销"""
        if self.editor and self.editor.undo():
            self.update_image_display()

    def on_redo(self):
        """重做"""
        if self.editor and self.editor.redo():
            self.update_image_display()

    def save_data(self):
        """保存修改后的分割结果"""
        if not self.editor:
            QMessageBox.warning(self, "提示", "请先加载影像！")
            return
        save_path, _ = QFileDialog.getSaveFileName(self, "保存分割结果", "", "NIfTI Files (*.nii.gz)")
        if not save_path:
            return
        try:
            self.editor.save_edited_seg(save_path)
            QMessageBox.information(self, "成功", "分割结果保存成功！")
        except Exception as e:
            QMessageBox.critical(self, "错误", f"保存失败：{str(e)}")

    # ========== 鼠标交互 ==========
    def on_mouse_press(self, event):
        """鼠标按下"""
        if not self.editor or (not self.draw_btn.isChecked() and not self.erase_btn.isChecked()):
            return
        self.is_drawing = True
        self.last_pos = event.pos()
        # 单次点击绘制
        self._draw_pixel(event.pos())

    def on_mouse_move(self, event):
        """鼠标移动（绘制线条）"""
        if not self.is_drawing or not self.editor:
            return
        # 计算鼠标移动路径，逐点绘制
        self._draw_pixel(event.pos())
        self.last_pos = event.pos()

    def on_mouse_release(self, event):
        """鼠标释放"""
        self.is_drawing = False

    def _draw_pixel(self, pos):
        """将鼠标位置转换为影像像素坐标，修改分割掩码"""
        # 获取显示的图片尺寸
        pixmap = self.edit_label.pixmap()
        if not pixmap:
            return
        # 计算鼠标位置对应的影像像素坐标（适配缩放）
        img_h, img_w = self.editor.orig_data[:, :, self.editor.current_slice].shape
        display_w = pixmap.width()
        display_h = pixmap.height()
        # 等比例缩放后的坐标
        x = int(pos.x() / display_w * img_w)
        y = int(pos.y() / display_h * img_h)
        # 确保坐标在有效范围内
        x = max(0, min(x, img_w-1))
        y = max(0, min(y, img_h-1))
        # 编辑分割掩码
        mode = "draw" if self.draw_btn.isChecked() else "erase"
        self.editor.edit_seg_mask(x, y, mode)
        # 更新显示
        self.update_image_display()
```

- **控制器层**：新增`SegmentationEditController.py`，将`SegmentationEditWidget`中的业务逻辑（如加载/保存）抽离到控制器，遵循MVC架构；
- **日志适配**：在`segmentation_editor.py`中加入`logging`模块，记录加载/修改/保存操作到`src/logs/app.log`；
- **异常处理**：补充更多异常捕获（如nii.gz文件损坏、尺寸不匹配、路径不存在等）；
- **配置适配**：将画笔默认大小、叠加颜色、透明度等参数抽离到`src/constant/`目录，便于配置管理。

#### 2. 功能拓展建议
- **多标签颜色自定义**：添加颜色选择器，为不同分割标签设置不同颜色；
- **批量填充/删除**：支持框选区域批量填充某标签或删除；
- **快捷键支持**：绑定画笔/橡皮擦切换（如B/E键）、撤销/重做（Ctrl+Z/Ctrl+Y）；
- **3D预览**：集成VTK实现分割掩码的3D体绘制，直观查看修改后的整体效果；
- **批量处理**：支持批量导入多个nii.gz分割结果，批量修改/保存。

### 总结
1. 核心流程：**加载nii.gz（原始+分割）→ 切片同步显示 → 画笔/橡皮擦修改分割掩码 → 保存新的nii.gz**；
2. 关键技术：`nibabel`解析nii.gz、Qt鼠标交互实现绘制、空间参数对齐确保导出结果可用；
3. 架构适配：代码按仓库分层设计（核心算法/UI组件/工具函数），可无缝集成到现有项目。

你可先运行`SegmentationEditWidget.py`测试基础功能，再逐步集成到仓库的主程序中，优先验证“加载→修改→保存”的核心流程，再迭代拓展功能。
